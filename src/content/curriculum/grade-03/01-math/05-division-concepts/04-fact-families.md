questions:
  - type: multiple-choice
    prompt: Which division fact belongs with 5 x 6 = 30?
    choices:
      - 30 / 6 = 5
      - 30 / 5 = 7
      - 6 / 5 = 30
      - 30 + 6 = 36
    correctAnswer: 30 / 6 = 5
  - type: text-input
    prompt: If 8 x 4 = 32, type the answer to 32 / 8.
    acceptedAnswers:
      - "4"
    answerType: number
  - type: fill-blank
    prompt: 7 x 3 = 21, so 21 / 3 = ___.
    sentenceBefore: 7 x 3 = 21, so 21 / 3 =
    sentenceAfter: .
    choices:
      - "3"
      - "7"
      - "18"
      - "24"
    correctAnswer: "7"
  - type: match-pairs
    prompt: Match each multiplication fact to a related division fact.
    pairs:
      - left: 9 x 5 = 45
        right: 45 / 5 = 9
      - left: 6 x 4 = 24
        right: 24 / 6 = 4
      - left: 8 x 7 = 56
        right: 56 / 8 = 7
  - type: order-items
    prompt: Order the missing answers from least to greatest.
    items:
      - 6 x 6 = 36 so 36 / 6 = ?
      - 3 x 8 = 24 so 24 / 8 = ?
      - 4 x 7 = 28 so 28 / 4 = ?
    correctOrder:
      - 3 x 8 = 24 so 24 / 8 = ?
      - 6 x 6 = 36 so 36 / 6 = ?
      - 4 x 7 = 28 so 28 / 4 = ?
  - type: multiple-choice
    prompt: Which number is in the fact family for 4, 9, and 36?
    choices:
      - "13"
      - "32"
      - "36"
      - "45"
    correctAnswer: "36"
  - type: text-input
    prompt: If 12 / 3 = 4, type the related multiplication product 3 x 4.
    acceptedAnswers:
      - "12"
    answerType: number
  - type: multiple-choice
    prompt: Which equation does not belong with 2, 8, and 16?
    choices:
      - 2 x 8 = 16
      - 8 x 2 = 16
      - 16 / 2 = 8
      - 16 / 8 = 8
    correctAnswer: 16 / 8 = 8
