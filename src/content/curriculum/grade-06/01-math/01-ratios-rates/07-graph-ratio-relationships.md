    prompt: In a ratio graph, equivalent ratios often fall on one straight ___.
    sentenceBefore: In a ratio graph, equivalent ratios often fall on one straight
    sentenceAfter: .
    choices:
      - line
      - circle
      - angle
      - fraction bar
    correctAnswer: line
  - type: match-pairs
    prompt: Match each point to the rate y per x.
    pairs:

        right: "6"

        right: "5"

        right: "3"
  - type: order-items
    prompt: Order the points by x-value.
    items:
      - (3, 15)

      - (3, 15)

    answerType: number
  - type: multiple-choice
