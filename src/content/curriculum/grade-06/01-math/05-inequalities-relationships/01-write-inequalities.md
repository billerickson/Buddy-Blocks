---
id: lesson_grade6_math_inequalities_write
slug: write-inequalities
title: Write Inequalities
xpBase: 10
questions:
  - type: multiple-choice
    prompt: Which inequality means "x is greater than 12"?
    choices:
      - x > 12
      - x < 12
      - x = 12
      - 12 > x
    correctAnswer: x > 12
  - type: text-input
    prompt: Type the symbol for "less than or equal to".
    acceptedAnswers:
      - <=
      - ≤
    answerType: text
  - type: fill-blank
    prompt: At least 8 means ___ 8.
    sentenceBefore: At least 8 means
    sentenceAfter: "8."
    choices:
      - greater than or equal to
      - less than
      - equal to only
      - divided by
    correctAnswer: greater than or equal to
  - type: match-pairs
    prompt: Match each phrase to an inequality.
    pairs:
      - left: no more than 5
        right: x <= 5
      - left: fewer than 9
        right: x < 9
      - left: at least 3
        right: x >= 3
  - type: order-items
    prompt: Order these boundaries from least to greatest.
    items:
      - x > 10
      - x > 2
      - x > 7
    correctOrder:
      - x > 2
      - x > 7
      - x > 10
  - type: multiple-choice
    prompt: A ride requires riders to be at least 48 inches tall. Which inequality fits height h?
    choices:
      - h >= 48
      - h <= 48
      - h < 48
      - h = 48 only
    correctAnswer: h >= 48
  - type: text-input
    prompt: Tickets cost $6 each and you can spend at most $30. Type the greatest number of tickets.
    acceptedAnswers:
      - "5"
    answerType: number
  - type: multiple-choice
    prompt: Which phrase matches n < 20?
    choices:
      - n is less than 20
      - n is more than 20
      - n is exactly 20
      - n is at least 20
    correctAnswer: n is less than 20
