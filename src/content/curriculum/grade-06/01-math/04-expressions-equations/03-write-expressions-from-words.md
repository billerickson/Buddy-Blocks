id: lesson_grade6_math_write_expressions_words
slug: write-expressions-from-words
title: Write Expressions From Words
xpBase: 10
questions:
  - type: multiple-choice
    prompt: Which expression means "9 less than x"?
    choices:
      - x - 9
      - 9 - x
      - 9x
      - x + 9
    correctAnswer: x - 9
  - type: text-input
    prompt: Type an expression for 4 times n using a multiplication symbol.
    acceptedAnswers:
      - 4 x n
      - n x 4
    answerType: text
  - type: fill-blank
    prompt: The phrase "the sum of a and 6" means a ___ 6.
    sentenceBefore: The phrase "the sum of a and 6" means a
    sentenceAfter: "6."
    choices:
      - +
      - "-"
      - x
      - /
    correctAnswer: +
  - type: match-pairs
    prompt: Match each phrase to an expression.
    pairs:
      - left: half of p
        right: p / 2
      - left: 7 more than y
        right: y + 7
      - left: the product of 3 and k
        right: 3k
  - type: order-items
    prompt: Order the words to match x + 12.
    items:
      - "12"
      - x
      - plus
    correctOrder:
      - x
      - plus
      - "12"
  - type: multiple-choice
    prompt: Which expression means "the quotient of m and 5"?
    choices:
      - m / 5
      - 5 / m
      - m + 5
      - 5m
    correctAnswer: m / 5
  - type: text-input
    prompt: Type an expression for 11 fewer than b.
    acceptedAnswers:
      - b - 11
    answerType: text
  - type: multiple-choice
    prompt: Which phrase matches 2(n + 4)?
    choices:
      - twice the sum of n and 4
      - two more than n and 4
      - n times 4 plus 2
      - half of n plus 4
    correctAnswer: twice the sum of n and 4
