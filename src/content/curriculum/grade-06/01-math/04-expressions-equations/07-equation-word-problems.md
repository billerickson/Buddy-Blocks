questions:
  - type: passage-question
    prompt: Read and model.
    passageTitle: Savings
    passage: Maya has $18 after saving $6 each week for w weeks.
    question: Which equation can find w?
    choices:
      - 6w = 18
      - w + 6 = 18
      - 18w = 6
      - 18 - 6 = w
    correctAnswer: 6w = 18
  - type: multiple-choice
    prompt: A number plus 14 equals 31. Which equation fits?
    choices:
      - x + 14 = 31
      - 14x = 31
      - x - 14 = 31
      - 31 + x = 14
    correctAnswer: x + 14 = 31
  - type: text-input
    prompt: "Solve the story: 5 bags have equal mass and total 22.5 kg. Type kg per bag."
    acceptedAnswers:
      - "4.5"
    answerType: number
  - type: fill-blank
    prompt: A one-step equation uses one inverse operation to isolate the ___.
    sentenceBefore: A one-step equation uses one inverse operation to isolate the
    sentenceAfter: .
    choices:
      - variable
      - title
      - unit rate only
      - coordinate plane
    correctAnswer: variable
  - type: match-pairs
    prompt: Match each story to an equation.
    pairs:
      - left: $4 per ticket costs $28
        right: 4t = 28
      - left: A height plus 3 equals 15
        right: h + 3 = 15
      - left: A rope cut by 2.5 leaves 7.5
        right: r - 2.5 = 7.5
  - type: order-items
    prompt: Order the steps for an equation story.
    items:
      - Solve and check

      - Define the variable
      - Write the equation
      - Solve and check
  - type: text-input
    prompt: A runner goes 2.5 miles each lap and runs 10 miles. Type the number of laps.
    acceptedAnswers:
      - "4"
    answerType: number
  - type: multiple-choice
    prompt: Which solution makes 3x = 21 true in a word problem?

      - x = 18
